  <script>
    // JavaScript ساده برای شبیه‌سازی روند (بدون ارسال/ذخیره)
    (function(){
      const options = document.querySelectorAll('.option');
      const btn = document.getElementById('requestBtn');
      const progressArea = document.getElementById('progressArea');
      const progressText = document.getElementById('progressText');
      const resultArea = document.getElementById('resultArea');
      const resultText = document.getElementById('resultText');
      const history = document.getElementById('history');

      let selectedCP = 160;
      options.forEach(o=>{
        o.addEventListener('click', ()=> {
          options.forEach(x=>x.classList.remove('selected'));
          o.classList.add('selected');
          selectedCP = parseInt(o.getAttribute('data-cp'),10);
        });
      });

      function safeFormatEmail(e){
        if(!e) return '—';
        // فقط برای نمایش: مخفی کردن بخشی از ایمیل
        const at = e.indexOf('@');
        if(at<=1) return '***' + e.slice(at);
        return e[0] + '***' + e.slice(at-1);
      }

      btn.addEventListener('click', ()=> {
        // هشدار دربارهٔ ورود اطلاعات حساس
        const email = document.getElementById('email').value.trim();
        const pass = document.getElementById('pass').value.trim();

        // هیچ داده‌ای ارسال نمی‌شود؛ اگر رمز وارد شده باشد، به کاربر هشدار می‌دهیم
        if(pass.length>0){
          alert('توجه: وارد کردن رمز واقعی در این صفحه توصیه نمی‌شود. این فقط یک دِمو است و رمزها ذخیره نمی‌شوند، اما بهتر است رمز واقعی را وارد نکنید.');
        }

        progressArea.style.display = 'block';
        resultArea.style.display = 'none';
        let seconds = 10;
        progressText.innerText = `در حال پردازش... ${seconds} ثانیه مانده`;
        btn.disabled = true;

        const timer = setInterval(()=> {
          seconds--;
          if(seconds<=0){
            clearInterval(timer);
            progressArea.style.display = 'none';
            resultArea.style.display = 'block';
            resultText.innerText = `✅ موفق — ${selectedCP} CP (نمایشی) به حساب شما اضافه شد. (${safeFormatEmail(email)})`;
            btn.disabled = false;

            // به‌صورت محلی فقط برای نمایش، تاریخچه را نشان می‌دهیم (در حافظهٔ مرورگر ذخیره نمی‌کنیم)
            const now = new Date();
            history.innerText = `${now.toLocaleString('fa-IR')} — درخواست ${selectedCP} CP برای ${safeFormatEmail(email)} (فقط نمایشی)`;
          } else {
            progressText.innerText = `در حال پردازش... ${seconds} ثانیه مانده`;
          }
        }, 1000);
      });
    })();
  </script>
